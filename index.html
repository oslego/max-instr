function getPieChartSlice(cx, cy, startAngle, endAngle, color){
  var slice = new createjs.Shape()

  // Convert degrees of an angle to radians
  function degToRad(deg){
  return (Math.PI / 180) * deg
  }

  slice
    .graphics
    .beginFill(color)
    .moveTo(cx, cy)
    .arc(cx, cy, 100, degToRad(startAngle), degToRad(endAngle))
    .lineTo(cx, cy)

  return slice
}

/* Draw a pie chart from the data in the activities array */
function drawPieChart(){
  var lastAngle = 0

  for ( var i = 0; i < activities.length; i++ ){

  var startAngle = lastAngle
  var endAngle = lastAngle + 
    (activities[i].hours / 24) * 360

  lastAngle = endAngle

  var slice = getPieChartSlice(100, 100, startAngle, endAngle, activities[i].color)
    stage.addChild(slice)
    stage.update()
  }
}

/* Draw the chart's legend with colors and text labels.*/
function drawLegend(){

  for ( var i = 0; i < activities.length; i++ ){

  /* Calculate the Y position of labels. */
  var yPos = i * 50 + 75 

  var colorBox = new createjs.Shape()
  colorBox
    .graphics
    .beginFill(activities[i].color)
    
  colorBox
    .graphics
    .drawRect(250, yPos, 20, 20)
    
  var label = new createjs.Text(activities[i].type)
  label.font = 'bold 18px Arial'
  label.x = 280
  label.y = yPos

  stage.addChild(label, colorBox)
  stage.update()
  }
}
